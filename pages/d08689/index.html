    单例模式

    # 单例模式

    # 定义

    • 单例模式：保证仅有一个实例，并提供一个访问它的全局访问点。

    # 实现

    # 1.创建对象与赋值
    // 创建对象与赋值
let Singlton = function (name) {
    this.name = name
}
Singlton.prototype.getName = function () {
    return this.name
}

    # 2.判断实例
    // 判断实例
    let ProxyCreateSinglton = (function () {
        let instance = null
        return function (name) {
            if (instance) {
                return instance
            }
            console.log('name', name);
            return instance = new Singlton(name)
        }
    })()
